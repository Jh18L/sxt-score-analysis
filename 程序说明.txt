一、程序介绍：
特点：网页程序，优美的现代UI界面，合理的交互设计，流畅的过度动画，清晰的文件结构，适应手机和电脑端
说明：模拟手机应用“生学堂”发送请求获取成绩数据并提供分析的网页程序。用户登录授权，模拟用户登录，获取数据，存储数据。
二、详细功能及原理：

        function generateReport(data) {
            const exam = data.data[0];
            const date = new Date().toLocaleString();
            
            // 计算分数线等级
            function getGradeLevel(score, details) {
                if (!details || details.length === 0) return "";
                
                // 按分数线从高到低排序
                const sortedLines = [...details].sort((a, b) => b.lineScore - a.lineScore);
                
                for (const line of sortedLines) {
                    if (score >= line.lineScore) {
                        return line.lineName;
                    }
                }
                
                return "Y";
            }
            
            // 生成分数线HTML
            function generateScoreLines(details) {
                if (!details || details.length === 0) return "";

                const sortedLines = [...details].sort((a, b) => b.lineScore - a.lineScore);

                    return `<div class="score-line">${line.lineName} ${line.lineScore}分（${line.scoreLineNum}人）</div>`;
                }).join('');

独立函数版本（推荐）
如果您不想修改 String 原型，可以这样封装：

javascript
// 独立加密函数
function encryptAES(plainText) {
  const CryptoJS = window.CryptoJS;
  const key = CryptoJS.enc.Utf8.parse("JMybKEd6L1cVpw==");
  const srcs = CryptoJS.enc.Utf8.parse(plainText);
  
  return CryptoJS.AES.encrypt(srcs, key, {
    mode: CryptoJS.mode.ECB,
    padding: CryptoJS.pad.Pkcs7
  }).toString();
}

// 使用示例
console.log(encryptAES("123456")); // U2FsdGVkX1+...

// aes-node.js
const CryptoJS = require("crypto-js");

module.exports = {
  encrypt: (plainText) => {
    const key = CryptoJS.enc.Utf8.parse("JMybKEd6L1cVpw==");
    const srcs = CryptoJS.enc.Utf8.parse(plainText);
    
    return CryptoJS.AES.encrypt(srcs, key, {
      mode: CryptoJS.mode.ECB,
      padding: CryptoJS.pad.Pkcs7
    }).toString();
  }
}
